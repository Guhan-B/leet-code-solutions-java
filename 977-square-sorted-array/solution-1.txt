/**
 * Brute Force Solution - (Square + Sort)

 * Time Complexity  : O(n) + O(nlogn)
 * Space Complexity : O(1)
 */
class Solution {
    public int[] sortedSquares(int[] numbers) {
        for (int i = 0; i < numbers.length; i++) {
            numbers[i] *= numbers[i];
        }

        Arrays.sort(numbers);

        return numbers;
    }
}
