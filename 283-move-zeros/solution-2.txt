/**
 * Optimal Solution - (Two Pointer)
 *
 * Time Complexity  : O(n)
 * Space Complexity : O(1)
 */
class Solution {
    public void moveZeroes(int[] numbers) {
        int n = numbers.length, j = 0;

        for (int i = 0; i < n; i++)
            if (numbers[i] != 0)
                numbers[j++] = numbers[i];

        while (j < n) numbers[j++] = 0;
    }
}
