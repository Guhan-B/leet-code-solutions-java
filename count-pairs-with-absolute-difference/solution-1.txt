/**
 * Brute Force Solution - (Nested Loop)
 *
 * Time Complexity  : O(n^2)
 * Space Complexity : O(1)
 */
class Solution {
    public int countKDifference(int[] numbers, int k) {
        int n = numbers.length, count = 0;

        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                if (Math.abs(numbers[i] - numbers[j]) == k)
                    count++;

        return count;
    }
}
