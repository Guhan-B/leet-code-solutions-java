/**
 * Optimal Solution - (Two Pointer | Slow & Fast Pointer | Single Pass)
 *
 * Time Complexity  : O(n)
 * Space Complexity : O(1)
 */
class Solution {
    public void binarySort(int[] array) {
        int swap;

        for (int i = 0, j = 0; i < array.length; i++) {
            if (array[i] == 0) {
                swap = array[i];
                array[i] = array[j];
                array[j] = swap;

                j++;
            }
        }
    }
}
