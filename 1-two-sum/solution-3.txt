/**
 * Two Sum - Optimal Solution (Sorting + Two Pointers)
 *
 * NOTE : In this solution, we can only return a boolean value indicating
          whether a solution exists or not. The array will be sorted.
 *
 * Time Complexity  : O(n) + O(nlogn) = O(nlogn)
 * Space Complexity : O(1)
 *
 * https://www.geeksforgeeks.org/problems/key-pair5616/1
 */
class Solution {
    public boolean twoSum(int[] numbers, int target) {
        int n = numbers.length;
        int i = 0, j = n - 1;

        Arrays.sort(numbers);

        while (i < j) {
            int sum = numbers[i] + numbers[j];

            if (sum > target) {
                j--;
            } else if (sum < target) {
                i++;
            } else {
                return true;
            }
        }

        return false;
    }
}
